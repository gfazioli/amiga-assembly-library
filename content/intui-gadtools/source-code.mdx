	subq.w	#1,d7			* Check if it's a single Gadget...

*   RESULT

*

*

*	FreeAsmGList()

STDTAG	dc.l 0,BUTTAG

 	dc.l GT_Underscore,$5F

	move.l	d0,d5

	lea	STDTAG(pc),a0
